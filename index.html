Документация модулей

Модуль validateGroup.js
/** * Проверяет формат учебной группы (цифра + 3 заглавные буквы) * @param {string} group - Название учебной группы * @returns {boolean} - true если формат корректен */ function validateGroup(group) { const pattern = /^\d[А-Я]{3}$/; return pattern.test(group); }
Модуль taskStorage.js
/** * Сохраняет задачи в localStorage * @param {Array} tasks - Массив задач */ function saveTasks(tasks) { localStorage.setItem('tasks', JSON.stringify(tasks)); } /** * Загружает задачи из localStorage * @returns {Array} - Массив задач или пустой массив */ function loadTasks() { const data = localStorage.getItem('tasks'); return data ? JSON.parse(data) : []; }
